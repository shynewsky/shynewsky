# 완전이진트리

```python
# 가장 아래쪽 리프노드에 1000 이하의 숫자가 적히고
# 그 부모는 자식들의 합으로 이루어진다
# [L]번 노드에 저장된 값은?
'''
[4, 1], [5, 2], [3, 3] 이면
4번 노드에 1
5번 노드에 2
3번 노드에 3    
'''
# 총 노드 5개인 완전이진트리
# 빈 트리 만들고 [0, 0, 0, 0, 0, 0]
# 리프노드 등록 [0, 0, 0, 3, 1, 2]
# 역순으로 돌면서, 0일때 [i*2]+[i*2+1]
```
```python
import sys
sys.stdin = open('input.txt')

# 입력 --------------------------------------------------
T = int(input())
for t in range(1, T+1):
    # N 노드개수, M 리프노드개수, L 선택노드번호
    N, M, L = map(int, input().split())
    data = [list(map(int, input().split())) for _ in range(M)]

# 코드 --------------------------------------------------

    # 완전이진트리 생성 -----------------------------------
    tree = [0] * (N+1) # 1-indexed
    for idx, val in data:
        tree[idx] = val

    # 완전이진트리 채우기 -----------------------------------
    for i in range(N-M, 0, -1): # 역순회
        if tree[i] == 0:
            tree[i] += tree[i*2]
            if N >= i*2+1 :
                tree[i] += tree[i*2+1]

# 출력 ----------------------------------------------------
    print(f'#{t}', tree[L])
```
